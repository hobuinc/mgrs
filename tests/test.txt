>>> import mgrs
>>> latitude = 42.0
>>> longitude = -93.0

>>> m = mgrs.MGRS()
>>> c = m.toMGRS(latitude, longitude)
>>> print(c)
15TWG0000049776

>>> lat, lon = m.toLatLon(c)
>>> '%.6f' % lat
'41.999998'

>>> '%.6f' % lon
'-93.000000'
